BETWEEN… AND…, NOT

Created@April 20, 2023 9:48 AM

DayDay 2
SELECT
column_name1,
column_name2
FROM table_name
WHERE column_name1 BETWEEN value1 AND value2
  • used to filter a range of values
  • there is also NOT BETWEEN ... AND ... for excluding certain options

Example #1

SELECT
payment_id,
payment_date
FROM payment
WHERE amount BETWEEN 1.99 AND 6.99

Example #2

SELECT
payment_id,
payment_date
FROM payment
WHERE amount NOT BETWEEN 1.99 AND 6.99

Example #3

SELECT 
payment_id,
amount
FROM payment
WHERE payment_date BETWEEN '2020-01-24' AND '2020-01-26'

Example #4

SELECT *
FROM rental
WHERE rental_date BETWEEN '2005-05-24' AND '2005-05-26 23:00'
ORDER BY rental_date DESC

Example #5

SELECT COUNT(*) FROM payment
WHERE amount BETWEEN 1.99 AND 3.99
AND payment_date BETWEEN '2020-01-26' AND '2020-01-27 23:59'
