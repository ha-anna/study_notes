CAST(expression AS datatype(length)) // length is optional
  • converts a value (of any type) into a specified datatype

Data types

Example #1

SELECT
CAST(scheduled_arrival AS VARCHAR)
FROM flights

Example #2

SELECT
scheduled_arrival,
CAST(scheduled_arrival AS DATE)
FROM flights

Example #3

SELECT
COALESCE(CAST(actual_arrival-scheduled_arrival AS VARCHAR), 'not arrived')
FROM flights

Example #4

SELECT
LENGTH(CAST(actual_arrival AS VARCHAR))
FROM flights

Example #5

SELECT
ticket_no,
CAST(ticket_no AS bigint)
FROM tickets
